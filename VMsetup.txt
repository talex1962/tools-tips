SETTING UP A VM FOR A BACKEND
-----------------------------

1. Utilities - install first

sudo apt-get -y install python-pip unzip python-dev python-docutils

2. Git and LD repository

sudo apt-get -y install git
mkdir ld
cd ld
git clone https://talex1962@github.com/talex1962/ld.git
git push -u origin master

3. Django & Apache & MySQL

sudo su
apt-get -y install python-imaging python-pythonmagick python-markdown python-textile python-django
apt-get -y install apache2 apache2.2-common apache2-mpm-prefork apache2-utils libapache2-mod-wsgi
apt-get -y install libmysqlclient-dev libexpat1 ssl-cert
apt-get -y install sqlite3 sqlite3-doc

pip install django
pip install MySQL-python

4. web.py, SQLAlchemy and sqlite3

sudo su
apt-get -y install sqlite3 sqlite3-doc
pip install SQLAlchemy
pip install web.py

5. Final cleanup

apt-get -y update && apt-get -y upgrade && apt-get -y dist-upgrade
