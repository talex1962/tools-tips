SETTING UP A VM FOR A BACKEND
-----------------------------

1. Utilities and Git - install first

sudo apt-get -y install python-pip unzip python-dev python-docutils
sudo apt-get -y install git

2. LD source tree

# go to the root directory into which the subdirectory 'ld' will be created
git clone https://talex1962@github.com/talex1962/ld.git
cd ld
git push -u origin master

3. Django & Apache & MySQL

sudo apt-get -y install python-imaging python-pythonmagick python-markdown python-textile python-django
sudo apt-get -y install apache2 apache2-data apache2-mpm-prefork apache2-utils libapache2-mod-wsgi
sudo apt-get -y install libmysqlclient-dev libexpat1 ssl-cert
sudo apt-get -y install sqlite3 sqlite3-doc

sudo pip install django
sudo pip install MySQL-python
sudo pip install httplib2

4. web.py, SQLAlchemy and sqlite3

sudo apt-get -y install sqlite3 sqlite3-doc
sudo pip install SQLAlchemy
sudo pip install web.py

5. Final cleanup

sudo apt-get -y update && sudo apt-get -y upgrade && sudo apt-get -y dist-upgrade
